```yui

# 簡易スタックマシン
# 命令は [opcode, operand] の形の配列で表す。operand が不要な命令は 0 を入れる。
#
# 対応命令:
#   ["PUSH", n] -- スタックに n を積む
#   ["POP",  0] -- スタック頂上を捨てる
#   ["DUP",  0] -- スタック頂上を複製する
#   ["ADD",  0] -- 上 2 つを取り出して a+b を積む
#   ["SUB",  0] -- 上 2 つを取り出して a-b を積む
#   ["MUL",  0] -- 上 2 つを取り出して a*b を積む
#
# プログラム実行後、スタックの頂上を答えとして返す。
# (Yui の配列は要素削除ができないので、論理的なスタックの大きさを top で管理し、
#  pop は top を減らすだけにする)

run=入力programに対し{
   stack=[]
   top=0
   n=programの大きさ
   i=0
   n回くり返す{
      instr=program[i]
      op=instr[0]
      arg=instr[1]

      もしopが"PUSH"ならば{
         # 論理 top が配列実長より小さければ上書き、そうでなければ末尾に追加
         sz=stackの大きさ
         もしtopがszより小さいならば{
            stack[top]=arg
         }
         そうでなければ{
            stackにargを追加する
         }
         topを増やす
      }
      もしopが"POP"ならば{
         topを減らす
      }
      もしopが"DUP"ならば{
         t=差(top,1)
         x=stack[t]
         sz=stackの大きさ
         もしtopがszより小さいならば{
            stack[top]=x
         }
         そうでなければ{
            stackにxを追加する
         }
         topを増やす
      }
      もしopが"ADD"ならば{
         topを減らす
         b=stack[top]
         topを減らす
         a=stack[top]
         stack[top]=和(a,b)
         topを増やす
      }
      もしopが"SUB"ならば{
         topを減らす
         b=stack[top]
         topを減らす
         a=stack[top]
         stack[top]=差(a,b)
         topを増やす
      }
      もしopが"MUL"ならば{
         topを減らす
         b=stack[top]
         topを減らす
         a=stack[top]
         stack[top]=積(a,b)
         topを増やす
      }
      iを増やす
   }
   # スタックの頂上を結果として返す
   topを減らす
   stack[top]が答え
}

# テスト: 3 + 4 = 7
>>> run([["PUSH",3],["PUSH",4],["ADD",0]])
7
# テスト: (3 + 4) * 2 = 14
>>> run([["PUSH",3],["PUSH",4],["ADD",0],["PUSH",2],["MUL",0]])
14
# テスト: 10 - (2 + 3) = 5
>>> run([["PUSH",10],["PUSH",2],["PUSH",3],["ADD",0],["SUB",0]])
5
# テスト: DUP で 5 を複製して掛ける → 25
>>> run([["PUSH",5],["DUP",0],["MUL",0]])
25
# テスト: POP で途中の値を捨てる → 7
>>> run([["PUSH",7],["PUSH",99],["POP",0]])
7
# テスト: 単一 PUSH
>>> run([["PUSH",42]])
42
```

```yui
# 新しい数字キー直後かどうかを fresh フラグ (1/0) で管理する。

calc=入力keysに対し{
   display=0
   acc=0
   op="+"
   fresh=1
   n=keysの大きさ
   i=0
   n回くり返す{
      k=keys[i]
      もしkが"="ならば{
         もしopが"+"ならば{
            display=和(acc,display)
         }
         もしopが"*"ならば{
            display=積(acc,display)
         }
         fresh=1
      }
      そうでなければ{
         もしkが"+"ならば{
            acc=display
            op="+"
            fresh=1
         }
         そうでなければ{
            もしkが"*"ならば{
               acc=display
               op="*"
               fresh=1
            }
            そうでなければ{
               # 1 桁数字キー。fresh なら置き換え、そうでなければ 10 倍して足す
               d=整数化(k)
               もしfreshが1ならば{
                  display=d
                  fresh=0
               }
               そうでなければ{
                  display=和(積(display,10),d)
               }
            }
         }
      }
      iを増やす
   }
   displayが答え
}

# === テスト: 単純な足し算 ===
>>> calc(["7","+","3","="])
10
# === テスト: 2 桁の数字入力 ===
>>> calc(["1","2","+","3","="])
15
# === テスト: 掛け算 ===
>>> calc(["6","*","7","="])
42
# === テスト: 空の入力 ===
>>> calc([])
0
# === テスト: 数字のみ (等号なし) ===
>>> calc(["4","2"])
42
```

```yui
標準ライブラリを使う
# ツェラーの公式で曜日番号を求める
# 曜日番号: 0=土, 1=日, 2=月, 3=火, 4=水, 5=木, 6=金
# (ツェラーの公式の定義に合わせている)
#
# 公式 (グレゴリオ暦):
#   h = (q + floor(13*(m+1)/5) + K + floor(K/4) + floor(J/4) - 2*J) mod 7
#   q = 日, m = 月 (3..14, 1月と2月は前年の13,14月として扱う),
#   Y = 年 (m が 13,14 のときは前年), K = Y mod 100, J = floor(Y/100)

weekday=入力y,m,dに対し{
   # 1,2 月は前年の 13,14 月として扱う
   もしmが2以下ならば{
      m=和(m,12)
      y=差(y,1)
   }
   K=剰余(y,100)
   J=商(差(y,K),100)
   a=d
   b=商(積(13,和(m,1)),5)
   c=K
   e=商(K,4)
   f=商(J,4)
   g=積(0,J)
   gTwo=差(0,積(2,J))
   s=和(a,b)
   s=和(s,c)
   s=和(s,e)
   s=和(s,f)
   s=和(s,gTwo)
   # Yui の 剰余 は負の数に対して数学的な結果を返すので、7 を足して正にしてから取り直す
   h=剰余(和(剰余(s,7),7),7)
   hが答え
}

# 曜日番号から日本語名に変換
name_of_weekday=入力hに対し{
   result="?"
   もしhが0ならば{ result="土" }
   もしhが1ならば{ result="日" }
   もしhが2ならば{ result="月" }
   もしhが3ならば{ result="火" }
   もしhが4ならば{ result="水" }
   もしhが5ならば{ result="木" }
   もしhが6ならば{ result="金" }
   resultが答え
}

# === テスト: 既知の曜日 ===
# 2000-01-01 は土曜日
>>> weekday(2000,1,1)
0
# 2024-01-01 は月曜日
>>> weekday(2024,1,1)
2
# 2024-04-24 は水曜日
>>> weekday(2024,4,24)
4
# 1999-12-31 は金曜日
>>> weekday(1999,12,31)
6
# === テスト: 名前への変換 ===
>>> name_of_weekday(0)
"土"
>>> name_of_weekday(4)
"水"
>>> name_of_weekday(weekday(2024,1,1))
"月"
```
